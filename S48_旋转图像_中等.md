# 第48题 旋转图像

```javascript
/**
 * @param {number[][]} matrix
 * @return {void} Do not return anything, modify matrix in-place instead.
 */
var rotate = function(matrix) {
    var l = matrix.length;
    var temp = JSON.parse(JSON.stringify(matrix));
    matrix.forEach((row, i) => {
        row.forEach((v, j) => {
            matrix[j][i] = temp[i][j];
        });
    });
    matrix.forEach(row => {
        row.reverse();
    });
};
```
