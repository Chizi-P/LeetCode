# 第169题 多数元素

```javascript
/**
 * @param {number[]} nums
 * @return {number}
 */
var majorityElement = function(nums) {
    var candidate, count = 0;
    nums.forEach(e => {
        if (count == 0) {
            candidate = e;
        }
        count += e == candidate ? 1 : -1;
    });
    return candidate;
};
```
