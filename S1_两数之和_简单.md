# 第1题 两数之和

#### 解一

```javascript
/**
 * @param {number[]} nums
 * @param {number} target
 * @return {number[]}
 */
var twoSum = function(nums, target) {
    var l = nums.length;
    for (let i = 0; i < l; i++) {
        for (let j = i + 1; j < l; j++) {
            if (nums[i] + nums[j] == target) return [i, j];
        }
    }
};
```



#### 解二

```javascript
/**
 * @param {number[]} nums
 * @param {number} target
 * @return {number[]}
 */
var twoSum = function(nums, target) {
    for (var d, m = {}, i = 0; i < nums.length; i++) {
        if ((d = target - nums[i]) in m) return [m[d], i];
        m[nums[i]] = i;
    }
};
```
